##### 1、编写一个函数，输入两个数组，输出它们的交集。输出数组中不含重复的元素，元素排列顺序可随意。
``` bash
    function test (arr1, arr2) {
        let newArr = []
```

##### 2、判断两个字符串是否同构。
    比如 s = "egg", t = "add", s和t就是同构
``` bash
    let str1 = 'asccdssaa'
    let str2 = 'tdaaeddtt'

    function test(str) {
        let arr = []
        for(let char of str) {
            if (!arr.includes(char)) {
                arr.push(char)
                str = str.replace(new RegExp(char,'g'), arr.indexOf(char) + 1)
            }
        }
        return str
    }
    test(str1) === test(str2)
```

##### 4、实现浅拷贝、深拷贝
``` bash
    let obj = {
        name: 'yy',
        age: 13,
        like: ['sing', 'dance']
    }
    // 浅拷贝
    function shallowClone(obj) {
        let newObj = {}
        for(let key in obj) {
            newObj[key] = obj[key]
        }
        return newObj
    }
    shallowClone(obj)
    // 深拷贝
    function deepClone(obj) {
        let newObj = Array.isArray(obj) ? [] : {}
        for(let key in obj) {
            if (typeof obj[key] === 'object') {
                newObj[key] = deepClone(obj[key])
            } else {
                newObj[key] = obj[key]
            }
        }
        return newObj
    }
    let obj = {
        name: 'yy',
        age: 13,
        like: ['sing', 'dance']
    }
    deepClone(obj)
```

##### 5、实现防抖、节流
``` bash
    // 防抖
    function debounce(fn, time) {
        let timeId = null
        return () => {
            clearTimeout(timeId)
            timeId = setTimeout(() => {
                fn()
            }, time)
        }
    }
    document.body.onclick  = debounce(function() {
        console.log('节流')
    }, 1000)
    // 节流
    function throtte(fn, time) {
        let startTime = 0

        return () => {
            let endTime = Date.now()
            if (endTime - startTime > time) {
                fn()
                startTime = endTime
            }
        }
    }
    document.body.onclick  = throtte(function() {
        console.log('防抖')
    }, 1000)
```

##### 6、实现call
    参考文献：https://github.com/mqyqingfeng/Blog/issues/12
``` bash
    let test = {
        name: 'test'
    }
    let obj = {
        name: '哈哈',
        speak: function() {
            console.log(this.name, arguments)
        }
    }

    Function.prototype.myCall = function(context, ...args) {
        // _fn_是任意起的，之后会被删除
        let context = context || window;
        context._fn_ = this;  //在obj上添加_fn_属性，值是this(要调用此方法的那个函数对象)。
        context._fn_(...args); //在obj上调用函数,那函数的this值就是obj.
        delete context._fn_; 
    }
    // 原生js
    obj.speak.call(test, 1, 2, 3)
    // 自定义call
    obj.speak.myCall(test, 1, 2, 3)

    // 解题思路
    将test构造为如下样子
    let test = {
        name: 'test',
        _fn_: function() {
            console.log(this.name, arguments)
        }
    }
    1、将函数设为对象的属性
    2、执行该函数
    3、删除该属性
```

##### 7、实现apply（原理和call差不多）
``` bash
    let test = {
        name: 'test'
    }
    let obj = {
        name: '哈哈',
        speak: function() {
            console.log(this.name, arguments)
        }
    }

    Function.prototype.myApply = function(context, arr) {
        var context = context || window
        context._fn_ = this
        if (!arr || arr.length === 0) {
            context._fn_()
        } else {
            let params = []
            for(let i = 0 ; i<arr.length ; i++){
                params.push(arr[i]) ;
            }
            context._fn_(...params)
        }
        delete context._fn_
    }

    // 原生js
    obj.speak.apply(test, [1, 2, 3])
    // 自定义apply
    obj.speak.myApply(test, [1, 2, 3])
```


##### 8、实现bind
``` bash
    let test = {
        name: 'test'
    }
    let obj = {
        name: '哈哈',
        speak: function() {
            this.habit = 'shopping';
            console.log(this.name, arguments)
        }
    }

    Function.prototype.myBind = function(context) {
        var self = this
        // 接受bind函数的参数，在此处为1，2，3
        var args = Array.from(arguments).slice(1)
        let fBound = function() {
            // 接受bind返回的函数的参数，在此处为4，5
            var args1 = Array.from(arguments)
            // 当bind返回的函数作为构造函数使用时,要判断一下
            self.apply(this instanceof fBound ? this : context, args.concat(args1))
        }
        fBound.prototype = Object.create(this.prototype);
        return fBound
    }

    // 原生js
    var s = obj.speak.bind(test, 1, 2, 3)
    // 自定义bind
    var s = obj.speak.myBind(test, 1, 2, 3)
    s(4, 5)
    // 复杂的情况
    let s1 = new s()
```
