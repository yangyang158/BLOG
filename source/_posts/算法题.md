##### 1、编写一个函数，输入两个数组，输出它们的交集。输出数组中不含重复的元素，元素排列顺序可随意。
``` bash
    function test (arr1, arr2) {
        let newArr = []
        arr1.forEach(function(item){
            if (arr2.includes(item)) {
               newArr.push(item) 
            }
        })
        return newArr
    }
    test([1,2,3,3,6,8,0], [2,2,5,9,4,8])

```

##### 2、判断两个字符串是否同构。
    比如 s = "egg", t = "add", s和t就是同构
``` bash
    let str1 = 'asccdssaa'
    let str2 = 'tdaaeddtt'

    function test(str) {
        let arr = []
        for(let char of str) {
            if (!arr.includes(char)) {
                arr.push(char)
                str = str.replace(new RegExp(char,'g'), arr.indexOf(char) + 1)
            }
        }
        return str
    }
    test(str1) === test(str2)
```

##### 3、判断一个页面使用最多的3个标签及使用次数
``` bash
    function getTagPage(str) {
        let nodes = Array.from(document.getElementsByTagName('*'))
        let obj = {}
        nodes.forEach(node => {
            let tagName = node.tagName.toLowerCase()
            if (obj[tagName]) {
                obj[tagName] = obj[tagName] + 1
            } else {
                obj[tagName] = 1
            }
        })
        return obj
    }
    var result = getTagPage()
    var sortArr = Object.keys(result).sort(function(a, b){
        return result[b] - result[a]
    })
```
